using System.Collections;
using System.Collections.Generic;
using UnityEngine;
using TMPro;
using UnityEngine.SceneManagement;

public class GameManager : MonoBehaviour
{
    public List targetPrefebs; //liståˆ—è¡¨ å¯ä»¥ç”¨addï¼Œremoveæ–¹æ³•æ·»åŠ æˆ–åˆ é™¤
	  //public GameObject[] target;          //æ•°ç»„ å ç”¨å›ºå®šçš„å†…å­˜æ— æ³•æ·»åŠ

    public bool gameStarted;
    public TextMeshProUGUI scoreText;
    public GameObject gameOver;
    public GameObject mainMenu;

    IEnumerator SpawnTarget()
    {
        while (gameStarted)
        {
            yield return new WaitForSeconds(spawnRate);
            int index = Random.Range(0, targetPrefebs.Count);
            Instantiate(targetPrefebs[index]);
        }
    }

    public void UpdateSource(int scoreToAdd)
    {
        score += scoreToAdd;
        scoreText.text = "Score:" + score;
    }

    public void GameIsOver() {
        gameStarted = false;
        gameOver.SetActive(true);
    }

    public void RestartGame() 
    {
        SceneManager.LoadScene(SceneManager.GetActiveScene().name);
    }
    public void StartNewGame(int switchD) {
        spawnRate /= switchD;
        gameStarted = true;
        StartCoroutine(SpawnTarget());
        mainMenu.SetActive(false);
    }
}

using System.Collections;
using System.Collections.Generic;
using UnityEngine;

public class Target : MonoBehaviour
{
    private Rigidbody rb;
    GameManager gameManager;

    public int point;
    public float miniSpeed = 12;
    public float maxSpeed = 16;
    public float maxTorque = 10;
    public float xRange = 4;

    public ParticleSystem explosionParticle;
    void Start()
    {
        gameManager = FindObjectOfType();
        rb = GetComponent();
        rb.AddForce(RandomSpeed(), ForceMode.Impulse);
        rb.AddTorque(RandomTorque(), RandomTorque(), RandomTorque());
        transform.position = RandomSpawnPos();
    }

    public void OnTriggerEnter(Collider other)
    {
        Destroy(gameObject);
        if (!gameObject.CompareTag("Bad"))
        {
            gameManager.GameIsOver();
        }      
    }

    void OnMouseDown()
    {
        if (gameManager.gameStarted)
        {
            Instantiate(explosionParticle, transform.position, explosionParticle.transform.rotation);
            gameManager.UpdateSource(point);
            Destroy(gameObject);
        }

        
    }

    Vector3 RandomSpeed()
    {
        return Vector3.up * Random.Range(miniSpeed, maxSpeed); // å› ä¸ºä¹˜äº†vector3.up æ‰€ä»¥æœ‰ä¸ªåˆå§‹å€¼
    }

    Vector3 RandomSpawnPos()
    {
        return new Vector3(Random.Range(-xRange, xRange), -1f, 0);
    }

    float RandomTorque() {
        return Random.Range(-maxTorque, maxTorque);
    }
}
